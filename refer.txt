<script>
// 获取baseurl
const baseUrl = '';

// 周刊数据 - 使用自动加载器生成
let weeklyData = [];

// 初始化周刊数据
function initWeeklyData() {
    if (typeof generateWeeklyData === 'function') {
        const generatedData = generateWeeklyData();
        weeklyData = generatedData.map(weekly => ({
            title: weekly.title,
            date: weekly.date,
            excerpt: weekly.content,
            url: baseUrl + "/weekly/" + weekly.id + "/",
            cover: weekly.cover || baseUrl + "/assets/images/weekly/" + weekly.id + "-cover.jpg"
        }));
    } else {
        // 降级方案
        weeklyData = [
            {
                title: "第1期 - 新的开始",
                date: "2025-01-15",
                excerpt: "这是第一期周刊，记录每周的学习、思考与生活点滴。希望通过这种方式，记录成长的轨迹。",
                url: baseUrl + "/weekly/weekly-001/",
                cover: baseUrl + "/assets/images/weekly/weekly-001-cover.jpg"
            }
        ];
    }
}

// 分页配置
const itemsPerPage = 6;
let currentPage = 1;

// 渲染周刊网格
function renderWeeklyGrid(page = 1) {
    const startIndex = (page - 1) * itemsPerPage;
    const endIndex = startIndex + itemsPerPage;
    const pageData = weeklyData.slice(startIndex, endIndex);
    
    const grid = document.getElementById('weekly-grid');
    grid.innerHTML = pageData.map(item => `
        <a href="${item.url}" class="weekly-card">
            ${item.cover ? `<div class="weekly-cover"><img src="${item.cover}" alt="${item.title}" class="cover-image"></div>` : ''}
            <div class="weekly-info">
                <h3>${item.title}</h3>
                <div class="date">${item.date}</div>
                <p class="excerpt">${item.excerpt}</p>
            </div>
        </a>
    `).join('');
}

// 渲染分页
function renderPagination() {
    const totalPages = Math.ceil(weeklyData.length / itemsPerPage);
    const pagination = document.getElementById('pagination');
    
    if (totalPages <= 1) {
        pagination.style.display = 'none';
        return;
    }
    
    pagination.innerHTML = `
        <button onclick="changePage(${currentPage - 1})" ${currentPage === 1 ? 'disabled' : ''}>
            上一页
        </button>
        <span class="page-info">第 ${currentPage} 页，共 ${totalPages} 页</span>
        <button onclick="changePage(${currentPage + 1})" ${currentPage === totalPages ? 'disabled' : ''}>
            下一页
        </button>
    `;
}

// 切换页面
function changePage(page) {
    const totalPages = Math.ceil(weeklyData.length / itemsPerPage);
    if (page < 1 || page > totalPages) return;
    
    currentPage = page;
    renderWeeklyGrid(currentPage);
    renderPagination();
    
    // 滚动到顶部
    window.scrollTo({ top: 0, behavior: 'smooth' });
}

// 初始化
document.addEventListener('DOMContentLoaded', function() {
    initWeeklyData();
    renderWeeklyGrid(1);
    renderPagination();
});

// 加载自动加载器脚本
const autoLoaderScript = document.createElement('script');
autoLoaderScript.src = baseUrl + '/scripts/auto-loader.js';
autoLoaderScript.onload = function() {
    initWeeklyData();
    renderWeeklyGrid(1);
    renderPagination();
};
document.head.appendChild(autoLoaderScript);
</script>
